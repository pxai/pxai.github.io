package ejemplo;

import javax.microedition.midlet.*;
// este es necesario para Display, Form y Command
import javax.microedition.lcdui.*;

/**
 * @author pello altadill
 */
public class HolaMundo extends MIDlet {
    private Display pantalla;
    private Form formulario;
    private Command ordenSalida = new Command ("Salir", Command.EXIT, 1);

    /**
     * HolaMundo
     * Constructor
     */
    public HolaMundo ()
    {
        pantalla = Display.getDisplay (this);
        formulario = new Form ("System Properties");
    }

    /**
     * startApp
     * inicio de la aplicación
     */
    public void startApp() {
            formulario.append("Hola Mundo");
            pantalla.setCurrent(formulario);

    }

    public void pauseApp() {
    }

    /**
     * commandAction
     * captura la orden de cierre
     */
    public void commandAction (Command c, Displayable s) {
        if (c == ordenSalida) {
            destroyApp (false);
            notifyDestroyed ();
        }
    }

    public void destroyApp(boolean unconditional) {
    }
}
